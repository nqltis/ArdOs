 *  = Ctrl
 #  = Shift
 A - D = Functions
 0 - 9 = Values

Modes : 
 - File Browser > File Reader
 -              > Code editor
 -              > Executor


{
  *# > File Browser

  File Browser {
    A : Prev File (up)
    B : Next File (down)
    C : Prev Dir
    D : Next Dir / Read File / Edit Code
  }

  File Reader {
    A : Prev Line
    B : Next Line
    C : Close
    D : //
  }

  Code Editor {
    A : Prev Line
    B : Next Line
    C : Close
    D : Save
  }

  Code Executor {
    *#C : Stop
    C : (when paused) Play
    D : (when paused) Next step
  }
}
